#%% Guess the coefficients
num_coeffs = 7
A_guess = np.empty(num_coeffs)
B_guess = np.empty(num_coeffs)
for k in np.arange(1,num_coeffs,1):
    A_guess[k] = 2.*np.mean(np.cos(2.*np.pi*k*t/T) * y)
    B_guess[k] = 2.*np.mean(np.sin(2.*np.pi*k*t/T) * y)
    if(A_guess[k]>=1):
        print('.....................')
    print('k = ' + str(k) + ', A_k = ' + str(round(A_guess[k],3))) 
    if(A_guess[k]>=1):
        print('.....................')
    if(B_guess[k]>=1):
        print('.....................')
    print('k = ' + str(k) + ', B_k = ' + str(round(B_guess[k],3)))       
    if(B_guess[k]>=1):
        print('.....................')
